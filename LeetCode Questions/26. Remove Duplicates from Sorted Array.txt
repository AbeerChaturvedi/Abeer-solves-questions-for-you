class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        '''I tried to solve this with creating a separate list and using not in function which was not a good solution.
        as the time complexity it took was O(N^2) and space for creating was O(N).
        found optimal soultion by doing it with 2 pointers

        first check if there are numbers in the list or not'''
        if not nums:
            return 0
        j = 0 #initialize the first pointer
        for i in range(1, len(nums)):
            if nums[j] != nums[i]:
                j+=1
                nums[j] = nums[i] #overwrite on the current index with unique number
            else:
                continue #pass if they are equal and move the i index
        return j+1
