class Solution:
    def reverse(self, x: int) -> int:
        sign = -1 if x < 0 else 1
        x = abs(x)

        reversed_ = 0
        while x:
            reversed_ = (reversed_*10) + (x%10)
            x //= 10

        if reversed_ > 2 ** 31:
            return 0
            
        return sign*reversed_
