class Solution:
    def romanToInt(self, s: str) -> int:
        dictionaryForVerify = {'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
        answer = 0
        for i in range(len(s)):
            if i + 1 < len(s) and dictionaryForVerify[s[i]] < dictionaryForVerify[s[i+1]]:
                answer -= dictionaryForVerify[s[i]]
            else:
                answer += dictionaryForVerify[s[i]]
        return answer
