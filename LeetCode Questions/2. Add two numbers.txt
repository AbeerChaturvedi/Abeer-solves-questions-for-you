# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode() #create a list 
        answer = dummy #getting the head
        carry = 0
        
        while l1 or l2 or carry: # if any has value we need to add
            valueOne = l1.val if l1 else 0
            valueTwo = l2.val if l2 else 0
            total = valueOne+valueTwo+carry
            carry = total // 10 # gives us the carry
            answer.next = ListNode(total%10) # creates next link and adds number
            answer = answer.next # move forward in link
            if l1:
                l1 = l1.next
            if l2:
                l2 = l2.next

        return dummy.next # return the list
