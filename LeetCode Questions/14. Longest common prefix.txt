class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        ans = "" #empty string to add prefix to
        sortedString = sorted(strs) #sorting in small to big
        first = sortedString[0] #taking smallest
        last = sortedString[-1] #taking biggest
        for i in range(min(len(first),len(last))):
            if first[i] != last[i]:
                return ans
            else:
                ans += first[i]
        return ans
